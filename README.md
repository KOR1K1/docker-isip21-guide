### docker-compose.yml
```yml
version: '3'
services:
  php:
    build:
      context: .
      dockerfile: ./Dockerfile
    volumes:
      - "./php:/var/www/html"

  nginx:
    image: nginx:latest
    ports:
      - 80:80
    volumes:
      - "./php:/var/www/html"
      - "./docker/nginx/conf.d/:/etc/nginx/conf.d"

  mysql:
    image: mysql:8
    environment:
      - MYSQL_ROOT_PASSWORD=root
    volumes:
      - mysql-data:/var/lib/mysql

  phpmyadmin:
    image: phpmyadmin
    restart: always
    ports:
      - 1337:80
    environment: 
      - PMA_ARBITRARY=1
volumes:
  mysql-data:
    external: true
```
